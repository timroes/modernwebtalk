ECMAScript 6


import {Point2D} from 'geometry'

class Point3D extends Point2D {
  constructor(x, y, z) {
    super(x, y);
    this.z = z;
  }

  toString() {
    return "[" + x + "," + y + "," + y + "]";
  }

  static zeroPoint() {
    return new Point3D(0, 0, 0);
  }
}
